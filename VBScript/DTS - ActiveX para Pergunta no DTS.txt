Function Main()
	Const vbYes = 6
	Const vbYesNo = 4
	Dim lResult

	lResult = MsgBox("Do you wish to skip the next task?", vbYesNo)
	
	If lResult = vbYes Then
		MsgBox("True")
	Else
		MsgBox("False")
	End If
	
	Main = DTSTaskExecResult_Success
End Function
